# 基本使用

## 创建一个 Vue 应用

<!-- tabs:start -->
<!-- tab:vue2 -->
```js
import Vue from "vue";

import App from "./App.vue";
import router from "./router";
import store from "./store";

const app = new Vue({
  router,
  store,
  render: h => h(App),
}).$mount("#app");
```

<!-- tab:vue3 -->
```js
import { createApp } from "vue";

import App from "./App.vue";
import router from "./router";
import { createPinia } from "pinia";

const app = createApp(App)
  .use(router)
  .use(createPinia())
  .mount("#app");
```
