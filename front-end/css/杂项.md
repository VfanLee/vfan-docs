  content: '';
  display: block;
  clear: both;
  /* 兼容性 */

  visibility: hidden;
}

.clearfix::before,
.clearfix::after {
  content: '';
  display: table;
}

  clear: both;
  /* 兼容性 */
  height: 0;
  visibility: hidden;
}
