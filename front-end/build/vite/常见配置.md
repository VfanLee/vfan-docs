## 配置 alias 别名

参考于：[resolve.alias](https://cn.vitejs.dev/config/shared-options.html#resolve-alias)

<!-- tabs:start -->
<!-- tab:对象写法 -->
```js
resolve: {
  alias: {
    '@': fileURLToPath(new URL('./src', import.meta.url))
  }
}
```

<!-- tab:数组写法 -->
```js
resolve: {
  alias: [
    {
      find: '@',
      replacement: fileURLToPath(new URL('./src', import.meta.url))
    }
  ]
}
```
<!-- tabs:end -->

## 配置 css 预处理器的全局参数

参考于：[css.preprocessorOptions](https://cn.vitejs.dev/config/shared-options.html#css-preprocessoroptions)

<!-- tabs:start -->
<!-- tab:全局自定义变量 -->
```scss
// ./src/styles/variable.scss";
$primary-color: #42b883;
```

<!-- tab:element-plus 自定义变量 -->
```scss
// ./src/styles/element/var.scss";
@forward 'element-plus/theme-chalk/src/common/var.scss' with (
  $colors: (
    'primary': (
      'base': #42b883
    )
  )
);
```

<!-- tab:vite.config.js -->
```js
export default defineConfig({
    plugins: [
    vue(),
    Components({
      // 配置 element-plus 采用 sass 样式
      resolvers: [ElementPlusResolver({ importStyle: 'sass' })]
    })
  ],
  css: {
    preprocessorOptions: {
      scss: {
        // 导入全局样式
        additionalData: `
        @use ./src/styles/element/var.scss" as *;
        @use "./src/styles/variable.scss" as *;
        `
      }
    }
  }
})
```
<!-- tabs:end -->
