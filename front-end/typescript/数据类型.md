# typescript 数据类型

## 各类型概览

```ts
/* null */
const null1: null = null

/* undefined */
const undefined1: undefined = undefined

/* number */
const number1: number = 1
const number2: number = 5.3
const number3: number = -10

/* string */
const string1: string = 'hi'
const string2: string = "hi"
const string3: string = `hi`

/* boolean */
const boolean1: boolean = true
const boolean2: boolean = false

/* Symbol */
const symbol1: Symbol = Symbol()

/* 数组 */
const arr1: number[] = [1, 2, 3]
const arr2: Array<string> = ['a', 'b', 'c']
const arr3: (number | string)[] = [1, 2, 3, 'a', 'b', 'c'] // 联合类型

/* 函数 */
function f1(num1: number, num2: number): number {
    return num1 + num2
}
const f2 = (num1: number, num2: number): number => num1 + num2
function f3(str1: string): void {}
// 可选参数
function f4(num1?: number, num2?: number): void {
    return num1 + num2
}
```

## 类型别名

```ts
type MyType = boolean | number | string | (boolean | number | string)[]

const m1: MyType = true
const m2: MyType = 123
const m3: MyType = 'abc'
const m4: MyType = [true, 123, 'abc']
```
